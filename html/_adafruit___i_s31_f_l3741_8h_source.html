21 // RGB pixel color order permutations
22 typedef enum {
23  // Offset: R G B
24  IS3741_RGB = ((0 << 4) | (1 << 2) | (2)), // Encode as R,G,B
25  IS3741_RBG = ((0 << 4) | (2 << 2) | (1)), // Encode as R,B,G
26  IS3741_GRB = ((1 << 4) | (0 << 2) | (2)), // Encode as G,R,B
27  IS3741_GBR = ((2 << 4) | (0 << 2) | (1)), // Encode as G,B,R
28  IS3741_BRG = ((1 << 4) | (2 << 2) | (0)), // Encode as B,R,G
29  IS3741_BGR = ((2 << 4) | (1 << 2) | (0)), // Encode as B,G,R
30 } IS3741_order;

222 /* =======================================================================
223  So, IN PRINCIPLE, additional classes Adafruit_IS31FL3741_monoGFX and
224  Adafruit_IS31FL3741_monoGFX_buffered could go here for hypothetical
225  single-color "grayscale" matrices -- they'd be similar to the two
226  colorGFX classes above, but without inheriting ColorOrder. Since no
227  actual hardware along such lines currently exists, they are not
228  implemented, but this is where they'd be. I've got deadlines.
229 
230  Now we'll build on the classes above to create specific LED board
231  varieties. These "complete" items are then instantiated in user code.
232  There are two of each -- a direct (unbuffered) and buffered version.
233  It's done this way (rather than a single class with a buffer flag) to
234  avoid dynamic allocation -- object & buffer just go on heap or stack
235  as needed (the optional canvas in glasses is an exception).
236  =======================================================================*/
